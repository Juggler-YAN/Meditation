### Q1

整数溢出会从另一个端点开始继续计算，浮点数上溢会返回INF，浮点数下溢会损失浮点数精度。

```c
#include <stdio.h>

int main(void) {
    // 整数溢出
    int a = 2147483647;
    printf("%d\n", a);
    a += 1;
    printf("%d\n", a);
    a -= 1;
    printf("%d\n", a);
    // 浮点数上溢
    float b = 3.4e38f;
    printf("%e\n", b);
    b *= 10.0f;
    printf("%e\n", b);
    // 浮点数下溢
    float c = 0.1234e-38f;
    printf("%e\n", c);
    c /= 10.0f;
    printf("%e\n", c);
    return 0;
}
```
